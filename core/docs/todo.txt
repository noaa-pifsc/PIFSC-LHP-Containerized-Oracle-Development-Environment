git bash commands:


	# change into repo directory:
	cd /c/Users/USERNAME/Documents/Version\ Control/Git/PIFSC-Containerized-Oracle-Development-Environment/core/scripts/client_scripts



	# Dev:

		# build and deploy the development scenario of the CODE (local)
		bash ./client_execute_CODE_scripts.sh deploy dev local

		# build and deploy the development scenario of the CODE (server)
		bash ./client_execute_CODE_scripts.sh deploy dev server

		# establish the connection to the docker server mapping the default ports so those access can be access on the client machine (make sure there are no containers using localhost that will cause a conflict)
		ssh -N -L 1521:localhost:1521 -L 8181:localhost:8181 dev_docker


		# shutdown the container (local) and retain volume
		bash ./client_execute_CODE_scripts.sh shutdown dev local no


		# shutdown the container (local) and remove volumes
		bash ./client_execute_CODE_scripts.sh shutdown dev local yes

		# shutdown the container (server) and retain volume
		bash ./client_execute_CODE_scripts.sh shutdown dev server no


		# shutdown the container (server) and remove volumes
		bash ./client_execute_CODE_scripts.sh shutdown dev server yes


	# Test:

		# build and deploy the test scenario of the CODE (local)
		bash ./client_execute_CODE_scripts.sh deploy test local

		# build and deploy the test scenario of the CODE (server)
		bash ./client_execute_CODE_scripts.sh deploy test server

		# establish the connection to the docker server mapping the default ports so those access can be access on the client machine (make sure there are no containers using localhost that will cause a conflict)
		ssh -N -L 1521:localhost:1521 -L 8181:localhost:8181 dev_docker

		bash ./client_execute_CODE_scripts.sh shutdown test local no
